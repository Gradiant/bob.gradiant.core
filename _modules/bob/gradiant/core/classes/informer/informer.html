[docs]    def message(self, message, color=Colors.reset, prefix='', suffix=''):
        if self.verbose:
            self.__reset()
            sys.stdout.write('{}{}{} - {}{}{}'.format(prefix, color, self.title, message, Colors.reset, suffix))
            sys.stdout.flush()



[docs]    def progress(self, message, count, total, color=Colors.reset):
        if self.verbose:
            self.__reset()
            bar_len = 60
            filled_len = int(round(bar_len * count / float(total)))

            percents = round(100.0 * count / float(total), 1)
            bar = '=' * filled_len + '-' * (bar_len - filled_len)

            sys.stdout.write('{}{} - {} [{}]{}%\r'.format(color, self.title, message, bar, percents))

            if count >= total:
                sys.stdout.write('{}\n'.format(Colors.FG.black))
            sys.stdout.flush()
